================================================================================
HOW TO USE
================================================================================

1. ASSIGNMENT
   final deliveryService = DeliveryService();
   await deliveryService.assignDeliveryAgent(order);
   // Result: order.deliveryAgentId = 'agent_1' (REAL AGENT)

2. OTP GENERATION
   final verificationService = DeliveryVerificationService();
   final otp = await verificationService.generateAndStoreOTP(orderId);
   // Result: '123456' stored in order document

3. OTP VERIFICATION
   final verified = await verificationService.verifyDeliveryOTP(
     orderId: orderId,
     providedOTP: '123456',
     agentId: agentId,
     latitude: 28.6139,
     longitude: 77.2090,
   );
   // Result: true if match, false if mismatch

4. EVENT LOGGING
   await verificationService.logDeliveryEvent(
     orderId: orderId,
     agentId: agentId,
     eventType: 'delivered',
     notes: 'Successfully delivered',
   );
