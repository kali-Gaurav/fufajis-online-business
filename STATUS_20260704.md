  - `computeHash(CartItem)` → SHA256 hash

  ```dart
  GooglePlacesService _places = GooglePlacesService();
  
  TextField(
    onChanged: (query) async {
      final predictions = await _places.getAutocomplete(query);
      setState(() => _predictions = predictions);
    },
  )
  
  onSelect: (prediction) async {
    final details = await _places.getPlaceDetails(prediction.placeId);
    if (details != null) {
      setState(() {
        latitude = details.latitude;
        longitude = details.longitude;
        fullAddress = details.formattedAddress;
        district = details.district;
        village = details.village;
        pincode = details.pincode;

  }
  ```
